#include <iostream>
#include <vector>

using namespace std;

// تابع جمع ماتریس ها
vector<vector<int>> addMatrices(const vector<vector<int>>& matrix1, const vector<vector<int>>& matrix2, int n) {
    vector<vector<int>> result(n, vector<int>(n, 0));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            result[i][j] = matrix1[i][j] + matrix2[i][j];
        }
    }

    return result;
}

// تایع تفریق ماتریس ها
vector<vector<int>> minusMatrices(const vector<vector<int>>& matrix1, const vector<vector<int>>& matrix2, int n) {
    vector<vector<int>> result(n, vector<int>(n, 0));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            result[i][j] = matrix1[i][j] - matrix2[i][j];
        }
    }

    return result;
}

// تابع ضرب ماتریس ها
vector<vector<int>> multiplyMatrices(const vector<vector<int>>& matrix1, const vector<vector<int>>& matrix2, int n) {
    vector<vector<int>> result(n, vector<int>(n, 0));

    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            for (int k = 0; k < n; ++k) {
                result[i][j] += matrix1[i][k] * matrix2[k][j];
            }
        }
    }

    return result;
}

int main() {
    int n;
    cout << "Enter the number of rows and columns: ";
    cin >> n;

    vector<vector<int>> matrix1(n, vector<int>(n, 0));
    vector<vector<int>> matrix2(n, vector<int>(n, 0));

    // گرفتن اعداد ماتریس اول
    cout << "Enter elements of the first matrix:" << endl;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cout << "Enter element a" << i + 1 << j + 1 << " : ";
            cin >> matrix1[i][j];
        }
    }

    // گرفتن اعداد ماتریس دوم
    cout << "Enter elements of the second matrix:" << endl;
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            cout << "Enter element b" << i + 1 << j + 1 << " : ";
            cin >> matrix2[i][j];
        }
    }

    // جمع دو ماتریس
    vector<vector<int>> sum = addMatrices(matrix1, matrix2, n);
    // تفریق دو ماتریس
    vector<vector<int>> difference = minusMatrices(matrix1, matrix2, n);
    // ضرب دو ماتریس
    vector<vector<int>> product = multiplyMatrices(matrix1, matrix2, n);

    // نشان دادن حاصل سه عملیات در خروجی
    cout << endl << "Sum of the matrices:" << endl;
    for (const auto& row : sum) {
        for (int val : row) {
            cout << val << " ";
        }
        cout << endl;
    }

    cout << endl << "minus of the matrices:" << endl;
    for (const auto& row : difference) {
        for (int val : row) {
            cout << val << " ";
        }
        cout << endl;
    }

    cout << endl << "multiplication of the matrices:" << endl;
    for (const auto& row : product) {
        for (int val : row) {
            cout << val << " ";
        }
        cout << endl;
    }

    return 0;
}

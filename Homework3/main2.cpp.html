#include <iostream>

using namespace std;

// تابع برای جابجایی عناصر دو سطر در ماتریس
void swapRows(int **matrix, int row1, int row2, int n) {
    for (int i = 0; i < n; i++) {
        swap(matrix[row1][i], matrix[row2][i]);
    }
}

// تابع برای محاسبه ترانهاده ماتریس
void transposeMatrix(int **matrix, int n) {
    for (int i = 0; i < n; i++) {
        for (int j = i + 1; j < n; j++) {
            swap(matrix[i][j], matrix[j][i]);
        }
    }
}

// تابع برای محاسبه دترمینان ماتریس
int determinant(int **matrix, int n) {
    if (n == 1) {
        return matrix[0][0];
    }

    if (n == 2) {
        return matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0];
    }

    int det = 0;

    for (int i = 0; i < n; i++) {
        int sign = (i % 2 == 0) ? 1 : -1;

        // جابجایی عناصر سطرها برای ایجاد صفر در ستون مورد نظر
        swapRows(matrix, 0, i, n);

        // محاسبه دترمینان برای هر زیرماتریس
        int **submatrix = new int *[n - 1];
        for (int j = 0; j < n - 1; j++) {
            submatrix[j] = new int[n - 1];
        }

        for (int j = 1; j < n; j++) {
            for (int k = 0; k < n - 1; k++) {
                submatrix[j - 1][k] = matrix[j][k + 1];
            }
        }

        det += sign * matrix[0][i] * determinant(submatrix, n - 1);

        // بازگرداندن عناصر به حالت اولیه
        swapRows(matrix, 0, i, n);

        // آزاد کردن حافظه مختص زیرماتریس
        for (int j = 0; j < n - 1; j++) {
            delete[] submatrix[j];
        }
        delete[] submatrix;
    }

    return det;
}

int main() {
    int n;

    // دریافت اندازه ماتریس از کاربر
    cout << "Enter the size of the matrix: ";
    cin >> n;

    // ایجاد ماتریس
    int **matrix = new int *[n];
    for (int i = 0; i < n; i++) {
        matrix[i] = new int[n];
    }

    // دریافت اعداد ماتریس از کاربر
    cout << "Enter the elements of the matrix:" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << "Enter the element [" << i + 1 << "][" << j + 1 << "]: ";
            cin >> matrix[i][j];
        }
    }

    cout << "\n";

    // چاپ ماتریس
    cout << "Entered matrix:" << endl;
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }

    cout << "\n";

    // محاسبه و چاپ ترانهاده ماتریس
    cout << "Transpose of the matrix:" << endl;
    transposeMatrix(matrix, n);
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << matrix[i][j] << " ";
        }
        cout << endl;
    }

    cout << "\n";

    // محاسبه و چاپ دترمینان ماتریس
    cout << "Determinant of the matrix: " << determinant(matrix, n) << endl;

    return 0;
}
